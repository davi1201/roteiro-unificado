```typescript
type ModuleErrors = Record<SiengeModuleSlug, Record<string, { message?: string }>>

export function TorreSiengeSection({ tenantId }: { tenantId: string }) {
  const store = useFormStore(tenantId)
  const { control, watch, formState: { errors } } = useForm<TorreSiengeData>({
    resolver: zodResolver(torreSiengeSchema),
    defaultValues: (store.sectionData[TabKey.TorreSienge] ?? {}) as Partial<TorreSiengeData>,
    mode: 'onBlur',
  })
  useEffect(() => {
    const sub = watch((v) => store.updateSection(TabKey.TorreSienge, v as Record<string, unknown>))
    return () => sub.unsubscribe()
  }, []) // eslint-disable-line react-hooks/exhaustive-deps
  const moduleErrors = (errors.modules as unknown as ModuleErrors) ?? {}
  return (
    <form className="max-w-5xl" noValidate>
      {SIENGE_MODULES.map((module) => {
        const modErr = moduleErrors[module.slug] ?? {}
        return (
          <article key={module.slug} className="mb-4 rounded-md border border-gray-200 bg-white p-4">
            <h3 className="mb-3 text-sm font-semibold text-gray-900">{module.label}</h3>
            <div className="grid grid-cols-1 gap-3 md:grid-cols-2">
              <SelectField name={`modules.${module.slug}.contratado` as FieldPath<TorreSiengeData>} ... />
              {/* outras 3 colunas */}
            </div>
            <div className="mt-3">
              <TextareaField name={`modules.${module.slug}.observacoes` as FieldPath<TorreSiengeData>} ... />
            </div>
          </article>
        )
      })}
    </form>
  )
}
```
