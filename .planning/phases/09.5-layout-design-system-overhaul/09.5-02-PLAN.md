    import { AdminSidebar } from './AdminSidebar'
    import { AdminHeader } from './AdminHeader'

    import { cn } from '@/lib/utils'

    **2. Componente:**

    ```typescript
    export function AdminLayout() {
      const [collapsed] = useSidebarCollapsed()
      const location = useLocation()
      const orgDetailMatch = useMatch('/admin/orgs/:orgId')

      const breadcrumb = useMemo<BreadcrumbItem[]>(() => {
        if (orgDetailMatch) {
          const orgId = orgDetailMatch.params.orgId ?? ''
          return [
            { label: 'Admin', href: '/admin/dashboard' },
            { label: 'Organizações', href: '/admin/dashboard' },
            { label: `Org ${orgId.slice(0, 8)}` },
          ]
        }
        if (location.pathname.startsWith('/admin/dashboard')) {
          return [{ label: 'Admin' }, { label: 'Organizações' }]
        }
        return [{ label: 'Admin' }]
      }, [orgDetailMatch, location.pathname])

      return (
        <div className="min-h-screen bg-gray-50">
          <AdminSidebar />
          <div
            className={cn(
              'flex flex-col min-h-screen transition-[margin-left] duration-[250ms] ease-out',
              collapsed ? 'ml-[60px]' : 'ml-60'
            )}
          >
            <AdminHeader breadcrumb={breadcrumb} />
            <main className="flex-1 p-8">
              <Outlet />
            </main>
          </div>
        </div>
      )
    }
    ```

    **3. Decisões importantes:**

    - **Breadcrumb dinâmico:** usar `useMatch` para detectar `/admin/orgs/:orgId` e `location.pathname` para o dashboard. O label da org usa fallback `Org {orgId.slice(0,8)}` porque o nome real está em um `useQuery` dentro de `OrgDetail` — AdminLayout NÃO deve duplicar a query (anti-pattern: layout disparando fetch). Se quiser nome real, OrgDetail pode atualizar via outra estratégia em fase futura.
    - **Actions slot vazio por enquanto:** não passar `actions` para AdminHeader nesta fase — o slot fica disponível para uso futuro (ex: botão "Nova Organização" na rota `/admin/dashboard`). Tarefa de wiring de actions pode ficar para um plano futuro de melhoria UX; este plano só garante que o slot existe.
    - **Margin dinâmico:** classes mutually exclusive `ml-[60px]` vs `ml-60` (15rem = 240px). Transition em `margin-left` apenas (não `all`) para evitar re-render visual de elementos internos.
    - **REMOVER** `mt-14 ml-60 p-8` (estático antigo). A altura do header (`h-14` = 56px) é tratada via `sticky top-0` no AdminHeader — o `<main>` flui naturalmente abaixo.

    **4. NÃO USAR:**
    - `useState` para `collapsed` aqui — usar APENAS `useSidebarCollapsed()` (hook do plan 01 é a single source of truth).
    - Imports de tela específica (AdminDashboard, OrgDetail).
